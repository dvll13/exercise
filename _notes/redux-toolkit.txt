//postSlice.js
import { createSlice } from '@reduxjs/toolkit'

const postsSlice = createSlice({
  name: 'posts',
  initialState,
  reducers: {
    addPost: (state, action) => {
      state.push(action.payload)
    }
  }
})

export const selectPosts = (state) => state.posts
export const { addPost } = postsSlice.actions
